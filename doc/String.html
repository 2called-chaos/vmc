def ellipsisize(ledge=10,redge=ledge*2)
  return self if self.length < ledge+redge
  sledge = '.'*ledge
  sredge = '.'*redge
  mid_length = self.length - ledge - redge
  gsub(/(#{sledge}).{#{mid_length},}(#{sredge})/, '\1...\2')
end
